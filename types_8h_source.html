00036 /**
00037  * Cross-platform compatibility types for off_t / time_t
00038  *
00039  * NOTE: This needs to be in a public header so that both the library
00040  * implementation and client applications both agree on the same types.
00041  * Otherwise we get undefined behavior.
00042  *
00043  * Use the "best" types that each platform provides. Currently we truncate
00044  * these intermediate representations for compatibility with the git ABI, but
00045  * if and when it changes to support 64 bit types, our code will naturally
00046  * adapt.
00047  * NOTE: These types should match those that are returned by our internal
00048  * stat() functions, for all platforms.
00049  */
00050 #include <sys/types.h>
00051 
00052 #if defined(_MSC_VER)
00053 
00054 typedef __int64 git_off_t;
00055 typedef __time64_t git_time_t;
00056 
00057 #elif defined(__MINGW32__)
00058 
00059 typedef off64_t git_off_t;
00060 typedef time_t git_time_t;
00061 
00062 #else  /* POSIX */
00063 
00064 /* 
00065  * Note: Can't use off_t since if a client program includes <sys/types.h>
00066  * before us (directly or indirectly), they'll get 32 bit off_t in their client
00067  * app, even though /we/ define _FILE_OFFSET_BITS=64.
00068  */
00069 typedef long long git_off_t;
00070 typedef time_t git_time_t;
00071 
00072 #endif
